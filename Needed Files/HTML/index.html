const OWNER_EMAIL = "real.hypertech.inc@gmail.com"; function doGet(e) { return HtmlService.createTemplateFromFile('index').evaluate() .addMetaTag('viewport', 'width=device-width, initial-scale=1'); } function processAction(command, userEmail) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const adminSheet = ss.getSheetByName("Admins"); const admins = adminSheet.getRange("A:A").getValues().flat(); let role = "User"; if (userEmail === OWNER_EMAIL) role = "Owner"; else if (admins.includes(userEmail)) role = "Admin"; // Logic: Only Owners can DELETE. Admins can BAN. Users can only VIEW. if (role === "User" && (command.includes("BAN") || command.includes("DELETE"))) { return "ACCESS DENIED: Insufficient Permissions"; } ss.getSheetByName("Requests").appendRow([new Date(), userEmail, role, command]); return `Success: ${role} executed ${command}`; }
